#version 330 core
layout(location = 0) in vec3 pos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 tPos;
layout(std140) uniform matrices{
	mat4 viewM;
	mat4 projectionM;
};
uniform mat4 modelM;
out vec3 fNormal;
out vec3 wPos;
out vec2 fTPos;

void main(){
	gl_Position =projectionM*viewM*modelM*vec4(pos,1.0f);
	fNormal =normalize(vec3(modelM*vec4(normal,0.0f)));
	fTPos = tPos;
	wPos = vec3(modelM*vec4(pos,1.0f));
	 
}
